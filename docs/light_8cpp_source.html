40 // Helper to send a PUT request to update light state
41 Result<void> sendUpdate(const nlohmann::json& state_update) {
42 if (!bridge || !bridge->isAuthenticated()) {
43 return Result<void>(ErrorCode::AuthenticationRequired,
44 "Bridge not authenticated");
45 }
46
47 const auto& bridge_info = bridge->getInfo();
48 if (bridge_info.ip_address.empty() || id.empty()) {
49 return Result<void>(ErrorCode::InvalidParameter,
50 "Bridge IP or light ID not set");
51 }
52
53 try {
54 HttpClient client;
55 client.setVerifySsl(false);
56 client.setTimeout(std::chrono::milliseconds(5000));
57
58 std::string url = "https://" + bridge_info.ip_address +
59 "/clip/v2/resource/light/" + id;
60
61 std::map<std::string, std::string> headers;
62 headers["hue-application-key"] = bridge->getAuthenticationKey();
63
64 std::string body = json_utils::toString(state_update);
65 auto response = client.put(url, body, headers);
66
67 if (!response.isSuccess()) {
68 if (response.status_code == 401 || response.status_code == 403) {
69 return Result<void>(ErrorCode::AuthenticationFailed,
70 "Authentication failed");
71 }
72 return Result<void>(ErrorCode::NetworkError,
73 "HTTP request failed: " + response.error_message);
74 }
75
76 // Parse response to check for errors
77 auto json_response = json_utils::parse(response.body);
78
79 if (json_response.contains("errors") && json_response["errors"].is_array()) {
80 auto errors = json_response["errors"];
81 if (!errors.empty()) {
82 std::string error_desc = json_utils::getValueOr<std::string>(
83 errors[0], "description", "Unknown error");
84 return Result<void>(ErrorCode::InvalidRequest, error_desc);
85 }
86 }
87
88 return Result<void>();
89
90 }
91 catch (const JsonParseException& e) {
92 return Result<void>(ErrorCode::InvalidRequest,
93 std::string("JSON error: ") + e.what());
94 }
95 catch (const std::exception& e) {
96 return Result<void>(ErrorCode::UnknownError,
97 std::string("Error: ") + e.what());
98 }
99 }

143 bool Light::isOn() const {
144 std::function extractIsOn = [this](std::string state_json) -> std::optional<bool> {
145 if (!state_json.empty()) {
146 try {
147 auto state = json_utils::parse(state_json);
148 if (state.contains("on") && state["on"].is_object()) {
149 auto on_obj = state["on"];
150 if (on_obj.contains("on") && on_obj["on"].is_boolean()) {
151 return on_obj["on"].template get<bool>();
152 }
153 }
154 }
155 catch (...) {
156 // Fall through to refresh cache or return nullopt
157 }
158 }
159 return std::nullopt;
160 };
161
162
163 if (!pImpl->bridge) {
164 // No bridge - cannot get state, return false
165 return false;
166 }
167
168 // Ask bridge for light state (cache-first, API-fallback)
169 std::string state_json = pImpl->bridge->getLightState(pImpl->id, false);
170 auto is_on_opt = extractIsOn(state_json);
171 if (is_on_opt.has_value()) {
172 return is_on_opt.value();
173 }
174
175 // Try refreshing cache
176 state_json = pImpl->bridge->getLightState(pImpl->id, true);
177 is_on_opt = extractIsOn(state_json);
178 return is_on_opt.value_or(false);
179 }

442 void Light::updateFromJson(const nlohmann::json& json) {
443 // Extract basic properties
444 pImpl->id = json_utils::getValueOr<std::string>(json, "id", pImpl->id);
445
446 // Extract metadata
447 if (json.contains("metadata") && json["metadata"].is_object()) {
448 auto metadata = json["metadata"];
449 pImpl->name = json_utils::getValueOr<std::string>(metadata, "name", "");
450 }
451
452 // Extract brightness (dimming)
453 if (json.contains("dimming") && json["dimming"].is_object()) {
454 auto dimming = json["dimming"];
455 auto brightness_val = json_utils::getValue<double>(dimming, "brightness");
456 if (brightness_val.has_value()) {
457 pImpl->capabilities.brightness = true;
458 }
459 }
460
461 // Extract color (XY)
462 if (json.contains("color") && json["color"].is_object()) {
463 auto color = json["color"];
464 if (color.contains("xy") && color["xy"].is_object()) {
465 auto xy = color["xy"];
466 auto x = json_utils::getValue<double>(xy, "x");
467 auto y = json_utils::getValue<double>(xy, "y");
468 if (x.has_value() && y.has_value()) {
469 pImpl->capabilities.color = true;
470 }
471 }
472 }
473
474 // Extract color temperature
475 if (json.contains("color_temperature") && json["color_temperature"].is_object()) {
476 auto color_temp = json["color_temperature"];
477 auto mirek = json_utils::getValue<int>(color_temp, "mirek");
478 if (mirek.has_value()) {
479 pImpl->capabilities.color_temperature = true;
480 }
481 }
482
483 // Determine capabilities from what's present in the light data
484 // On/off is always available
485 pImpl->capabilities.on_off = true;
486
487 // Effects capability (check if effects key exists)
488 if (json.contains("effects")) {
489 pImpl->capabilities.effects = true;
490 }
491 }
