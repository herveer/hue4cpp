78 /**
79 * @brief Unified event args for StateManager::OnResourceEvent
80 *
81 * Carries all information about any resource change, addition, or removal
82 * fired by the StateManager. Inspect @c type to determine what happened
83 * and which fields are meaningful.
84 *
85 * | type | resource_id | state_json |
86 * |-------------------------|-------------|------------------|
87 * | LightStateChanged | light id | updated JSON |
88 * | LightAdded | light id | initial JSON |
89 * | LightRemoved | light id | empty |
90 * | SensorStateChanged | sensor id | updated JSON |
91 * | SensorAdded | sensor id | initial JSON |
92 * | SensorRemoved | sensor id | empty |
93 * | BridgeConnected | empty | empty |
94 * | BridgeDisconnected | empty | empty |
95 */
96 struct ResourceEventArgs {
97 EventType type; ///< What happened
98 std::string resource_id; ///< Affected resource id (empty for bridge events)
99 std::string state_json; ///< Resource state JSON (empty when not applicable)

102 : type(EventType::Unknown) {
103 }
104
105 ResourceEventArgs(EventType t,
106 const std::string& id,
107 const std::string& json = "")
108 : type(t), resource_id(id), state_json(json) {
109 }
