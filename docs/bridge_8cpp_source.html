53 Result<std::string> Bridge::authenticate(const std::string& app_name,
54 const std::string& device_name) {
55 if (_info.ip_address.empty()) {
56 return Result<std::string>(ErrorCode::InvalidParameter,
57 "Bridge IP address is not set");
58 }
59
60 if (app_name.empty()) {
61 return Result<std::string>(ErrorCode::InvalidParameter,
62 "Application name cannot be empty");
63 }
64
65 // Construct device type string
66 std::string devicetype = app_name;
67 if (!device_name.empty()) {
68 devicetype += "#" + device_name;
69 }
70
71 // Create HTTP client
72 HttpClient client;
73 client.setVerifySsl(false); // Hue bridges use self-signed certificates
74 client.setTimeout(AUTH_TIMEOUT);
75
76 // Construct the URL for authentication
77 std::string url = "https://" + _info.ip_address + "/api";
78
79 // Create JSON request body
80 nlohmann::json request_body;
81 request_body["devicetype"] = devicetype;
82 std::string json_str = json_utils::toString(request_body);
83
84 // Try to authenticate with retries
85 for (int retry = 0; retry < MAX_AUTH_RETRIES; ++retry) {
86 auto response = client.post(url, json_str);
87
88 if (!response.isSuccess()) {
89 // Network error
90 return Result<std::string>(ErrorCode::NetworkError,
91 "Failed to connect to bridge: " + response.error_message);
92 }
93
94 try {
95 auto json_response = json_utils::parse(response.body);
96
97 // The response is an array of status objects
98 if (!json_response.is_array() || json_response.empty()) {
99 return Result<std::string>(ErrorCode::InvalidRequest,
100 "Invalid response from bridge");
101 }
102
103 auto first_item = json_response[0];
104
105 // Check for success
106 if (first_item.contains("success")) {
107 auto success = first_item["success"];
108 if (success.contains("username")) {
109 std::string username = success["username"].get<std::string>();
110 _auth_key = username;
111 // Note: SSE connection must be manually started via getStateManager().start()
112 return Result<std::string>(username);
113 }
114 }
115
116 // Check for error
117 if (first_item.contains("error")) {
118 auto error = first_item["error"];
119 int error_type = json_utils::getValueOr<int>(error, "type", 0);
120 std::string error_desc = json_utils::getValueOr<std::string>(
121 error, "description", "Unknown error");
122
123 // Error type 101 means link button not pressed
124 if (error_type == 101) {
125 // Continue retrying
126 if (retry < MAX_AUTH_RETRIES - 1) {
127 std::this_thread::sleep_for(AUTH_RETRY_INTERVAL);
128 continue;
129 }
130 else {
131 return Result<std::string>(ErrorCode::AuthenticationFailed,
132 "Link button not pressed within timeout period. " + error_desc);
133 }
134 }
135
136 // Other errors are fatal
137 return Result<std::string>(ErrorCode::AuthenticationFailed, error_desc);
138 }
139
140 // Unknown response format
141 return Result<std::string>(ErrorCode::InvalidRequest,
142 "Unexpected response from bridge");
143
144 }
145 catch (const JsonParseException& e) {
146 return Result<std::string>(ErrorCode::InvalidRequest,
147 std::string("Failed to parse bridge response: ") + e.what());
148 }
149 catch (const std::exception& e) {
150 return Result<std::string>(ErrorCode::UnknownError,
151 std::string("Authentication error: ") + e.what());
152 }
153 }
154
155 // Should not reach here, but just in case
156 return Result<std::string>(ErrorCode::TimeoutError,
157 "Authentication timed out");
158 }

173 Result<void> Bridge::validateAuthentication() const {
174 if (_auth_key.empty()) {
175 return Result<void>(ErrorCode::AuthenticationRequired,
176 "No authentication key set");
177 }
178
179 if (_info.ip_address.empty()) {
180 return Result<void>(ErrorCode::InvalidParameter,
181 "Bridge IP address is not set");
182 }
183
184 try {
185 HttpClient client;
186 client.setVerifySsl(false); // Hue bridges use self-signed certificates
187 client.setTimeout(AUTH_TIMEOUT);
188
189 // Try to access the config endpoint with the authentication key
190 std::string url = "https://" + _info.ip_address +
191 "/clip/v2/resource/bridge";
192
193 std::map<std::string, std::string> headers;
194 headers["hue-application-key"] = _auth_key;
195
196 auto response = client.get(url, headers);
197
198 if (!response.isSuccess()) {
199 if (response.status_code == 401 || response.status_code == 403) {
200 return Result<void>(ErrorCode::AuthenticationFailed,
201 "Invalid authentication key");
202 }
203 return Result<void>(ErrorCode::NetworkError,
204 "Failed to validate authentication: " + response.error_message);
205 }
206
207 // Parse response to check for errors
208 try {
209 auto json_response = json_utils::parse(response.body);
210
211 // Check if response contains an error
212 if (json_response.contains("errors") && json_response["errors"].is_array()) {
213 auto errors = json_response["errors"];
214 if (!errors.empty()) {
215 std::string error_desc = json_utils::getValueOr<std::string>(
216 errors[0], "description", "Unknown error");
217 return Result<void>(ErrorCode::AuthenticationFailed, error_desc);
218 }
219 }
220
221 // If we got here, authentication is valid
222 return Result<void>();
223
224 }
225 catch (const JsonParseException&) {
226 // If we can't parse the response but got a 200, assume it's valid
227 return Result<void>();
228 }
229
230 }
231 catch (const std::exception& e) {
232 return Result<void>(ErrorCode::UnknownError,
233 std::string("Validation error: ") + e.what());
234 }

551
552 std::map<std::string, std::string> headers;
553 headers["hue-application-key"] = _auth_key;
554
555 auto response = client.get(url, headers);
556
557 if (!response.isSuccess()) {
558 continue; // Try next resource type
559 }
560
561 auto json_response = json_utils::parse(response.body);
562
563 // Check if response contains errors
