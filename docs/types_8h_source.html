50 /**
51 * @brief Create from Kelvin temperature

53 * @return ColorTemperature in mireds
54 * @throws std::invalid_argument if kelvin is out of range
55 */
56 static ColorTemperature fromKelvin(uint16_t kelvin) {
57 if (kelvin < 2000 || kelvin > 6500) {
58 throw std::invalid_argument("Kelvin temperature must be between 2000 and 6500");
