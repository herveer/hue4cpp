50 /**
51 * @brief Create from Kelvin temperature
52 * @param kelvin Temperature in Kelvin (2000-6500)
53 * @return ColorTemperature in mireds
54 * @throws std::invalid_argument if kelvin is out of range
55 */
56 static ColorTemperature fromKelvin(uint16_t kelvin) {
57 if (kelvin < 2000 || kelvin > 6500) {
58 throw std::invalid_argument("Kelvin temperature must be between 2000 and 6500");
59 }
60 return ColorTemperature(static_cast<uint16_t>(1000000 / kelvin));
61 }
62
63 /**
64 * @brief Convert to Kelvin temperature
65 * @return Temperature in Kelvin
66 * @throws std::invalid_argument if mireds is invalid
67 */
68 uint16_t toKelvin() const {
69 if (mireds == 0) {
70 throw std::invalid_argument("Mireds value cannot be zero");
71 }
72 return static_cast<uint16_t>(1000000 / mireds);
73 }
74 };

282 Result(ErrorCode err, const std::string& msg = "")
283 : error(err), error_message(msg) {
284 }
