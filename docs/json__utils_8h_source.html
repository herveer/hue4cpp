37 /**
38 * @brief Safely get a value from a JSON object
39 * @tparam T The type to extract
40 * @param json The JSON object
41 * @param key The key to look up
42 * @return Optional value if key exists and has correct type
43 */
44 template<typename T>
45 std::optional<T> getValue(const nlohmann::json& json, const std::string& key) {
46 if (!json.is_object() || !json.contains(key)) {
47 return std::nullopt;
48 }
49
50 try {
51 return json[key].get<T>();
52 } catch (const nlohmann::json::type_error&) {
53 // Type mismatch
54 return std::nullopt;
55 } catch (const nlohmann::json::exception&) {
56 // Other JSON errors
57 return std::nullopt;
58 }
59 }
