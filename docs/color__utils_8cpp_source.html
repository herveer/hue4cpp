18 return (value > 0.04045f) ?
19 std::pow((value + 0.055f) / 1.055f, 2.4f) :
20 (value / 12.92f);
21 };

25 b = gammaCorrect(b);

49RGBColor xyToRgb(const XYColor& xy, uint8_t brightness) {
50 // Calculate z from x and y
51 float z = 1.0f - xy.x - xy.y;
52
53 // Calculate Y from brightness (0-100 to 0-1)
54 float Y = brightness / 100.0f;
55
56 // Calculate X and Z from x, y, z and Y

58 float Z = (Y / xy.y) * z;

67 return (value <= 0.0031308f) ?
68 12.92f * value :
69 1.055f * std::pow(value, 1.0f / 2.4f) - 0.055f;
70 };

74 b = reverseGamma(b);

121 float b = rgb.b / 255.0f;
122

124 float min = std::min({r, g, b});

126

128 v = max * 100.0f;
129
130 // Calculate saturation
131 if (max < 0.0001f) {
132 s = 0.0f;
133 h = 0.0f;
134 return;
135 }
136 s = (delta / max) * 100.0f;
137
138 // Calculate hue
139 if (delta < 0.0001f) {
140 h = 0.0f;

142 h = 60.0f * std::fmod(((g - b) / delta), 6.0f);
143 } else if (max == g) {
