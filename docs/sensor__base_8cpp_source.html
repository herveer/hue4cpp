126 void Sensor::onResourceEvent(const ResourceEventArgs& e) {
127 // Only react to sensor state changes that concern this specific sensor.
128 if (!e.isSensorEvent() || e.type != EventType::SensorStateChanged || e.resource_id != _id) {
129 return;
130 }
131
132 // Fire the generic per-sensor event so any external subscriber is notified.
133 OnStateChanged.Notify(e);
134
135 // Let the derived class fire its typed property notifications.
136 try {
137 auto delta = nlohmann::json::parse(e.state_json);
138
139 if (delta.contains("enabled")) {
140 auto val = json_utils::getValueOr<bool>(delta, "enabled", _enabled);
141 SetPropertyValueAndNotify<&Sensor::Enabled>(_enabled, val);
142 }
143
144 // Metadata rename pushed from the bridge.
145 if (delta.contains("metadata") && delta["metadata"].is_object()) {
146 const auto& meta = delta["metadata"];
147 if (meta.contains("name") && meta["name"].is_string()) {
148 auto newName = meta["name"].get<std::string>();
149 SetPropertyValueAndNotify<&Sensor::Name>(_name, newName);
150 }
151 }
152
153 notifyStateProperties(delta);
154 }
155 catch (...) {
156 // Malformed delta — derived class will handle its own fallback.
157 }
158 }

160 void Sensor::sendUpdate(const nlohmann::json& state_update) {
161 if (!_bridge || !_bridge->isAuthenticated()) {
162 throw BridgeNotReachableException("Bridge not authenticated");
163 }
164
165 const auto& bridge_info = _bridge->getInfo();
166 if (bridge_info.ip_address.empty() || _id.empty()) {
167 throw InvalidParameterException("Bridge IP or sensor ID not set");
168 }
169
170 try {
171 HttpClient client;
172 client.setVerifySsl(false);
173 client.setTimeout(std::chrono::milliseconds(5000));
174
175 std::string url = "https://" + bridge_info.ip_address +
176 "/clip/v2/resource/" + getResourceTypeString() + "/" + _id;
177
178 std::map<std::string, std::string> headers;
179 headers["hue-application-key"] = _bridge->getAuthenticationKey();
180
181 std::string body = json_utils::toString(state_update);
182 auto response = client.put(url, body, headers);
183
184 if (!response.isSuccess()) {
185 if (response.status_code == 401 || response.status_code == 403) {
186 throw AuthenticationException("Authentication failed");
187 }
188 throw NetworkException("HTTP request failed: " + response.error_message);
189 }
190
191 auto json_response = json_utils::parse(response.body);
192 if (json_response.contains("errors") && json_response["errors"].is_array()) {
193 const auto& errors = json_response["errors"];
194 if (!errors.empty()) {
195 throw InvalidParameterException(
196 json_utils::getValueOr<std::string>(errors[0], "description", "Unknown error"));
197 }
198 }
199 }
200 catch (const HueException&) { throw; }
201 catch (const std::exception& e) {
202 throw BridgeNotReachableException(std::string("Error: ") + e.what());
203 }
204 }

246 case SensorType::CameraMotion:
