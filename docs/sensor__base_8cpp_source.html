73 bool Sensor::isEnabled() const {
74 // Extract enabled state from cached JSON
75 std::function extractEnabled = [](const std::string& state_json) -> std::optional<bool> {
76 if (!state_json.empty()) {
77 try {
78 auto state = json_utils::parse(state_json);
79 if (state.contains("enabled") && state["enabled"].is_boolean()) {
80 return state["enabled"].template get<bool>();
81 }
82 }
83 catch (...) {
84 // Fall through to refresh cache or return default
85 }
86 }
87 return std::nullopt;
88 };
89
90 if (!_bridge) {
91 return false;

93
94 std::string state_json = _bridge->getSensorState(_id, getResourceTypeString(), false);
95 auto enabled_opt = extractEnabled(state_json);
96 if (enabled_opt.has_value()) {
97 return enabled_opt.value();
98 }
99
100 state_json = _bridge->getSensorState(_id, getResourceTypeString(), true);
101 enabled_opt = extractEnabled(state_json);
102 return enabled_opt.value_or(false);

145 void Sensor::onResourceEvent(const ResourceEventArgs& e) {
146 // Only react to sensor state changes that concern this specific sensor.
147 if (!e.isSensorEvent() || e.type != EventType::SensorStateChanged || e.resource_id != _id) {
148 return;
149 }
150
151 // Fire the generic per-sensor event so any external subscriber is notified.

154 // Let the derived class fire its typed property notifications.

156 auto delta = nlohmann::json::parse(e.state_json);
157 notifyStateProperties(delta);

159 catch (...) {
