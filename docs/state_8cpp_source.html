174 void StateManager::updateFromEvent(const std::string& event_json) {
175 if (event_json.empty()) {
176 return;
177 }
178
179 try {
180 auto json = nlohmann::json::parse(event_json);
181
182 if (!json.is_array()) {
183 return;
184 }
185
186 for (const auto& event_item : json) {
187 if (!event_item.contains("data") || !event_item.is_object()) {
188 continue;
189 }
190
191 std::string event_type = event_item.value("type", "");
192 const auto& data_array = event_item["data"];
193
194 if (!data_array.is_array()) {
195 continue;
196 }
197
198 for (const auto& resource : data_array) {
199 if (!resource.contains("id") || !resource.contains("type")) {
200 continue;
201 }
202
203 std::string resource_id = resource["id"];
204 std::string resource_type = resource["type"];
205
206 if (event_type == "delete") {
207 std::lock_guard<std::mutex> lock(_state_mutex);
208 _resource_states.erase(resource_id);
209 }
210 else {
211 mergeResourceState(resource_id, resource.dump());
212 }
213
214 if (resource_type == "light") {
215 if (event_type == "add") {
216 OnResourceEvent.Notify(ResourceEventArgs(EventType::LightAdded, resource_id, resource.dump()));
217 }
218 else if (event_type == "delete") {
219 OnResourceEvent.Notify(ResourceEventArgs(EventType::LightRemoved, resource_id));
220 }
221 else {
222 OnResourceEvent.Notify(ResourceEventArgs(EventType::LightStateChanged, resource_id, resource.dump()));
223 }

225 else if (resource_type == "motion" || resource_type == "temperature" ||
