49 // Merge JSON delta into existing state
50 void mergeResourceState(const std::string& resource_id, const nlohmann::json& delta) {
51 std::lock_guard<std::mutex> lock(state_mutex);
52
53 auto it = resource_states.find(resource_id);
54 if (it != resource_states.end()) {
55 // Existing state - merge delta
56 try {
57 auto existing = nlohmann::json::parse(it->second);
58 existing.merge_patch(delta);
59 it->second = existing.dump();
60 }
61 catch (...) {
62 // If parse fails, replace with delta
63 resource_states[resource_id] = delta.dump();
64 }
65 }
66 else {
67 // No existing state - store delta as initial state
68 resource_states[resource_id] = delta.dump();
69 }
70 }

86 Result<void> StateManager::start() {
87 if (pImpl->running) {
88 return Result<void>(ErrorCode::InvalidRequest, "StateManager already running");
89 }
90
91 if (!pImpl->bridge) {
92 return Result<void>(ErrorCode::InvalidRequest, "No bridge associated with StateManager");
93 }
94
95 // Get bridge info for SSE endpoint
96 const auto& bridge_info = pImpl->bridge->getInfo();
97 if (bridge_info.ip_address.empty()) {
98 return Result<void>(ErrorCode::InvalidRequest, "Bridge IP address not set");
99 }
100
101 const std::string auth_key = pImpl->bridge->getAuthenticationKey();
102 if (auth_key.empty()) {
103 return Result<void>(ErrorCode::AuthenticationRequired, "Bridge not authenticated");
104 }
105
106 // Create SSE endpoint URL
107 const std::string sse_url = "https://" + bridge_info.ip_address + "/eventstream/clip/v2";
108
109 // Create and configure SSE client
110 pImpl->sse_client = std::make_unique<SSEClient>(sse_url);
111 pImpl->sse_client->setAuthHeader("hue-application-key", auth_key);
112 pImpl->sse_client->setVerifySsl(false); // Hue bridges use self-signed certificates
113 pImpl->sse_client->setReconnection(true);
114
115 // Set up event callback
116 pImpl->sse_client->onEvent([this](const SSEEvent& sse_event) {
117 // Process SSE event and update state
118 updateFromEvent(sse_event.data);
119 });
120
121 // Set up connection state callback
122 pImpl->sse_client->onConnectionChange([this](bool connected) {
123 // Clear cache on disconnect/reconnect
124 clearCache();
125 if (connected) {
126 Event event(EventType::BridgeConnected, "", "");
127 pImpl->notifyCallbacks(event);
128 }
129 else {
130 Event event(EventType::BridgeDisconnected, "", "");
131 pImpl->notifyCallbacks(event);
132 }
133 });
134
135 // Connect to SSE stream
136 auto result = pImpl->sse_client->connect();
137 if (!result) {
138 pImpl->sse_client.reset();
139 return result;
140 }
141
142 pImpl->running = true;

204 void StateManager::updateFromEvent(const std::string& event_json) {
205 if (event_json.empty()) {
206 return;
207 }
208
209 try {
210 // Parse the event JSON
211 auto json = nlohmann::json::parse(event_json);
212
213 // SSE events from Hue API V2 have this structure:
214 // [
215 // {
216 // "creationtime": "2023-...",
217 // "data": [ { "id": "...", "type": "light", ... } ],
218 // "id": "...",
219 // "type": "update"
220 // }
221 // ]
222
223 if (!json.is_array()) {
224 return;
225 }
226
227 for (const auto& event_item : json) {
228 if (!event_item.contains("data") || !event_item.is_object()) {
229 continue;
230 }
231
232 std::string event_type = event_item.value("type", "");
233 const auto& data_array = event_item["data"];
234
235 if (!data_array.is_array()) {
236 continue;
237 }
238
239 // Process each resource in the event
240 for (const auto& resource : data_array) {
241 if (!resource.contains("id") || !resource.contains("type")) {
242 continue;
243 }
244
245 std::string resource_id = resource["id"];
246 std::string resource_type = resource["type"];
247
248 // Update internal state cache with delta merge
249 if (event_type == "delete") {
250 // Remove from cache on delete
251 std::lock_guard<std::mutex> lock(pImpl->state_mutex);
252 pImpl->resource_states.erase(resource_id);
253 }
254 else {
255 // Merge delta into existing state for add/update events
256 pImpl->mergeResourceState(resource_id, resource);
257 }
258
259 // Handle light state updates
260 if (resource_type == "light") {
261 // Determine event type
262 EventType evt_type = EventType::LightStateChanged;
263 if (event_type == "add") {
264 evt_type = EventType::LightAdded;
265 }
266 else if (event_type == "delete") {
267 evt_type = EventType::LightRemoved;
268 }
269
270 // Notify callbacks with current state
271 std::string current_state = getResourceState(resource_id);
272 Event event(evt_type, resource_id, resource.dump());
273 pImpl->notifyCallbacks(event);
274 }
275 // Handle sensor state updates (motion, temperature, light_level, button)
276 else if (resource_type == "motion" || resource_type == "temperature" ||
277 resource_type == "light_level" || resource_type == "button") {
278 // Determine event type
279 EventType evt_type = EventType::SensorStateChanged;
280 if (event_type == "add") {
281 evt_type = EventType::SensorAdded;
282 }
283 else if (event_type == "delete") {
284 evt_type = EventType::SensorRemoved;
285 }
286
287 // Notify callbacks with event data
288 Event event(evt_type, resource_id, resource.dump());
289 pImpl->notifyCallbacks(event);
290 }
291 // TODO: Add more resource types here (rooms, zones, scenes, etc.)
292 }
293 }
294 }
295 catch (const std::exception& e) {
296 std::cout << "StateManager::updateFromEvent: Failed to parse event JSON " << e.what() << std::endl;
297 }
