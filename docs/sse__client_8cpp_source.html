80 * @param line Line from SSE stream
81 * @param current_event Current event being built
82 * @return true if event is complete, false otherwise

84 bool parseSSELine(const std::string& line, SSEEvent& current_event) {
85 if (line.empty()) {
86 // Empty line signals end of event
87 return !current_event.data.empty();
88 }
89
90 if (line[0] == ':') {
91 // Comment line, ignore
92 return false;
93 }
94
95 size_t colon_pos = line.find(':');
96 if (colon_pos == std::string::npos) {
97 // Field with no value
98 return false;
99 }
100
101 std::string field = line.substr(0, colon_pos);
102 std::string value = line.substr(colon_pos + 1);
103
104 // Skip leading space in value
105 if (!value.empty() && value[0] == ' ') {
106 value = value.substr(1);
107 }
108
109 if (field == "event") {
110 current_event.event_type = value;
111 } else if (field == "data") {
112 if (!current_event.data.empty()) {
113 current_event.data += "\n";
114 }
115 current_event.data += value;
116 } else if (field == "id") {
117 current_event.id = value;
118 }
119 // retry field is ignored for now
120
121 return false;
122 }

124 void connectionLoop() {
125 std::chrono::seconds current_retry_delay = reconnect_initial_delay;
126
127 while (should_run) {
128 try {
129 // Prepare headers
130 cpr::Header headers;
131 if (!auth_header_name.empty() && !auth_header_value.empty()) {
132 headers[auth_header_name] = auth_header_value;
133 }
134 headers["Accept"] = "text/event-stream";
135 headers["Cache-Control"] = "no-cache";
136
137 // Set up session
138 cpr::Session session;
139 session.SetUrl(cpr::Url{url});
140 session.SetHeader(headers);
141 session.SetTimeout(timeout);
142 session.SetVerifySsl(cpr::VerifySsl{verify_ssl});
143
144 // Create a custom write callback to process SSE stream
145 std::string buffer;
146 SSEEvent current_event;
147
148 std::function<bool(std::string_view, intptr_t)> write_callback =
149 [&](std::string_view data, intptr_t) -> bool {
150 buffer.append(data.data(), data.size());
151
152 // Process complete lines
153 size_t newline_pos;
154 while ((newline_pos = buffer.find('\n')) != std::string::npos) {
155 std::string line = buffer.substr(0, newline_pos);
156 buffer = buffer.substr(newline_pos + 1);
157
158 // Remove \r if present
159 if (!line.empty() && line.back() == '\r') {
160 line.pop_back();
161 }
162
163 // Parse line
164 if (parseSSELine(line, current_event)) {
165 // Event complete, notify callback
166 notifyEvent(current_event);
167 current_event = SSEEvent();
168 }
169 }
170
171 return should_run; // Continue while should_run is true
172 };
173
174 session.SetWriteCallback(cpr::WriteCallback{write_callback});
175
176 // Notify connected
177 notifyConnectionChange(true);
178 current_retry_delay = reconnect_initial_delay; // Reset retry delay on successful connection
179
180 // Perform the request (this blocks until connection ends)
181 auto response = session.Get();
182
183 // Connection ended
184 notifyConnectionChange(false);
185
186 // Check if we should reconnect
187 if (!should_run) {
188 break;
189 }
190
191 if (!reconnection_enabled) {
192 break;
193 }
194
195 } catch (const std::exception& e) {
196 notifyConnectionChange(false);
197
198 if (!should_run || !reconnection_enabled) {
199 break;
200 }
201 }
202
203 // Wait before reconnecting
204 if (should_run && reconnection_enabled) {
205 std::this_thread::sleep_for(current_retry_delay);
206
207 // Exponential backoff
208 if (current_retry_delay == std::chrono::seconds(0)) current_retry_delay = std::chrono::seconds(1);
209 current_retry_delay = (std::min)(current_retry_delay * RECONNECT_BACKOFF_MULTIPLIER, reconnect_max_delay);
210 }
211 }
212 }
